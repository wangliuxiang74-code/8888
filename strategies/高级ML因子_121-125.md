## 121. 强化学习做市策略

```python
import numpy as np
import random

class RLMarketMaker:
    """强化学习做市商"""
    
    def __init__(self):
        self.q_table = {}  # Q表
        self.gamma = 0.9   # 折扣因子
        self.alpha = 0.1    # 学习率
        self.epsilon = 0.1  # 探索率
        
    def get_state(self, order_book, inventory, vol):
        """状态: 订单簿深度 + 库存 + 波动率"""
        bid_depth = sum(order_book['bid'][:5])
        ask_depth = sum(order_book['ask'][:5])
        return f"{bid_depth}_{inventory}_{vol}"
    
    def get_action(self, state):
        """动作: 报价/不报价"""
        if random.random() < self.epsilon:
            return random.choice(['bid', 'ask', 'hold'])
        return self.q_table.get(state, {}).get('hold', 'hold')
    
    def update_q(self, state, action, reward, next_state):
        """Q学习更新"""
        old_q = self.q_table.get(state, {}).get(action, 0)
        max_next = max(self.q_table.get(next_state, {}).values(), default=0)
        new_q = old_q + self.alpha * (reward + self.gamma * max_next - old_q)
        
        if state not in self.q_table:
            self.q_table[state] = {}
        self.q_table[state][action] = new_q
    
    def run(self, price_series, order_book_series):
        """运行做市策略"""
        for i in range(100, len(price_series)):
            state = self.get_state(
                order_book_series[i],
                inventory=0,  # 简化
                vol=np.std(price_series[i-20:i])
            )
            
            action = self.get_action(state)
            
            # 计算 reward (价差收益 - 库存风险)
            reward = self.calculate_reward(action, price_series[i])
            
            next_state = self.get_state(
                order_book_series[i+1] if i+1 < len(order_book_series) else order_book_series[i],
                inventory=0, vol=np.std(price_series[i-19:i+1])
            )
            
            self.update_q(state, action, reward, next_state)
```

## 122. 时序卷积网络 (TCN)

```python
import torch
import torch.nn as nn

class TCNModel(nn.Module):
    """时序卷积网络"""
    
    def __init__(self, input_size, num_channels=[64, 128, 256], kernel_size=3):
        super().__init__()
        
        layers = []
        num_levels = len(num_channels)
        
        for i in range(num_levels):
            in_channels = input_size if i == 0 else num_channels[i-1]
            out_channels = num_channels[i]
            
            # 膨胀卷积
            dilation = 2 ** i
            padding = (kernel_size - 1) * dilation
            
            layers.append(nn.Conv1d(
                in_channels, out_channels, kernel_size,
                padding=padding, dilation=dilation
            ))
            layers.append(nn.ReLU())
            layers.append(nn.BatchNorm1d(out_channels))
        
        self.network = nn.Sequential(*layers)
        self.fc = nn.Linear(num_channels[-1], 1)
    
    def forward(self, x):
        # x: (batch, seq_len, features)
        x = x.transpose(1, 2)  # (batch, features, seq_len)
        x = self.network(x)
        x = x[:, :, -1]  # 取最后一个时间步
        return self.fc(x)

# 使用示例
def predict_with_tcn(model, features):
    """用TCN预测"""
    model.eval()
    with torch.no_grad():
        x = torch.FloatTensor(features).unsqueeze(0)
        pred = model(x)
    return pred.item()
```

## 123. BERT新闻分类

```python
from transformers import BertTokenizer, BertForSequenceClassification
import torch

class BertNewsClassifier:
    """BERT新闻分类器"""
    
    def __init__(self, model_path=None):
        self.tokenizer = BertTokenizer.from_pretrained('bert-base-uncased')
        
        if model_path:
            self.model = BertForSequenceClassification.from_pretrained(model_path)
        else:
            # 初始化预训练模型
            self.model = BertForSequenceClassification.from_pretrained(
                'bert-base-uncased',
                num_labels=3  # 利好/利空/中性
            )
        
        self.model.eval()
        
        self.label_map = {0: 'bearish', 1: 'neutral', 2: 'bullish'}
    
    def predict(self, text):
        """预测新闻情绪"""
        inputs = self.tokenizer(
            text, 
            return_tensors='pt',
            truncation=True,
            max_length=512
        )
        
        with torch.no_grad():
            outputs = self.model(**inputs)
            probs = torch.softmax(outputs.logits, dim=1)
            pred = torch.argmax(probs, dim=1).item()
        
        return {
            'label': self.label_map[pred],
            'confidence': probs[0][pred].item(),
            'probs': {
                'bearish': probs[0][0].item(),
                'neutral': probs[0][1].item(),
                'bullish': probs[0][2].item()
            }
        }
    
    def batch_predict(self, news_list):
        """批量预测"""
        return [self.predict(news) for news in news_list]
```

```python
from gensim.models import Word2Vec
import numpy as np

class GoldWord2Vec:
    """黄金新闻词向量模型"""
    
    def __init__(self):
        self.model = None
        self.gold_embedding = None
        
    def train(self, news_corpus):
        """
        news_corpus: [['gold', 'rises', 'fed'], ['dollar', 'falls'], ...]
        """
        self.model = Word2Vec(
            sentences=news_corpus,
            vector_size=100,    # 词向量维度
            window=5,           # 上下文窗口
            min_count=2,        # 最小词频
            workers=4,          # 并行训练
            epochs=20           # 训练轮数
        )
        
        # 计算"黄金"的核心词向量
        if 'gold' in self.model.wv:
            self.gold_embedding = self.model.wv['gold']
    
    def similarity(self, word1, word2):
        """计算两个词的相似度"""
        if word1 in self.model.wv and word2 in self.model.wv:
            return self.model.wv.similarity(word1, word2)
        return 0.0
    
    def most_similar(self, word, topn=10):
        """找最相似的词"""
        if word in self.model.wv:
            return self.model.wv.most_similar(word, topn=topn)
        return []
    
    def get_news_sentiment(self, words):
        """根据词向量计算新闻情绪"""
        bullish_words = ['rise', 'gain', 'bullish', 'upgrade', 'buy', 'surge']
        bearish_words = ['fall', 'drop', 'bearish', 'downgrade', 'sell', 'decline']
        
        score = 0
        for word in words:
            if word in self.model.wv:
                for bw in bullish_words:
                    if bw in self.model.wv:
                        score += self.model.wv.similarity(word, bw)
                for bw in bearish_words:
                    if bw in self.model.wv:
                        score -= self.model.wv.similarity(word, bw)
        
        return score
```

## 125. STL时间序列分解

```python
from statsmodels.tsa.seasonal import STL
import pandas as pd
import numpy as np

class STLDecomposition:
    """STL时间序列分解"""
    
    def __init__(self, period=7, robust=True):
        """
        period: 周期 (7=一周, 30=一月)
        robust: 是否使用鲁棒估计
        """
        self.period = period
        self.robust = robust
    
    def decompose(self, price_series):
        """
        分解为: 趋势 + 季节 + 残差
        """
        # STL分解
        stl = STL(price_series, period=self.period, robust=self.robust)
        result = stl.fit()
        
        return {
            'trend': result.trend,
            'seasonal': result.seasonal,
            'residual': result.resid,
            'observed': result.observed
        }
    
    def get_signals(self, price_series):
        """生成交易信号"""
        decomp = self.decompose(price_series)
        
        # 趋势信号
        trend = decomp['trend']
        trend_derivative = trend.diff()
        
        # 季节性信号
        seasonal = decomp['seasonal']
        
        # 残差异常检测
        residual = decomp['residual']
        residual_std = residual.std()
        
        signals = []
        
        for i in range(len(price_series)):
            signal = 'HOLD'
            
            # 趋势反转
            if i > 0 and trend_derivative.iloc[i-1] < 0 and trend_derivative.iloc[i] > 0:
                signal = 'BUY'
            elif i > 0 and trend_derivative.iloc[i-1] > 0 and trend_derivative.iloc[i] < 0:
                signal = 'SELL'
            
            # 残差异常
            if abs(residual.iloc[i]) > 2 * residual_std:
                signal = 'VOLATILITY_ALERT'
            
            signals.append(signal)
        
        return signals
    
    def get_harmonic_ratio(self, price_series):
        """计算 harmonics ratio (季节性强度)"""
        decomp = self.decompose(price_series)
        
        seasonal_var = decomp['seasonal'].var()
        residual_var = decomp['residual'].var()
        
        if residual_var > 0:
            return seasonal_var / residual_var
        return 0

# 使用示例
def analyze_gold_price(prices):
    """分析黄金价格"""
    stl = STLDecomposition(period=7)  # 周周期
    
    # 分解
    decomp = stl.decompose(prices)
    
    # 信号
    signals = stl.get_signals(prices)
    
    # 季节性强度
    hr = stl.get_harmonic_ratio(prices)
    
    return {
        'trend': decomp['trend'].iloc[-1],
        'seasonal': decomp['seasonal'].iloc[-1],
        'residual': decomp['residual'].iloc[-1],
        'signal': signals[-1],
        'seasonal_strength': hr
    }
```

```python
        # 综合信号
        score = 0
        if results['bert']['label'] == 'bullish':
            score += 0.3
        elif results['bert']['label'] == 'bearish':
            score -= 0.3
        
        if results['stl']['trend'] > results['stl']['seasonal']:
            score += 0.2
        
        return {
            'score': score,
            'signal': 'BUY' if score > 0.3 else 'SELL' if score < -0.3 else 'HOLD',
            'details': results
        }
```
