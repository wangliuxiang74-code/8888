## 第一章：Python新闻抓取

### 1.1 RSS订阅
```python
import feedparser
import requests

class NewsFetcher:
    def __init__(self):
        self.sources = {
            'reuters': 'https://www.reutersagency.com/feed/?best-topics=commodities-metals',
            'gold': 'https://www.kitco.com/RSS/kitco-metadata-news.xml'
        }
    
    def fetch_rss(self, source):
        feed = feedparser.parse(self.sources[source])
        news = []
        
        for entry in feed.entries[:10]:
            news.append({
                'title': entry.title,
                'link': entry.link,
                'published': entry.published,
                'summary': entry.summary
            })
        
        return news
```

### 1.2 网页抓取
```python
from bs4 import BeautifulSoup
import requests

def scrape_news(url):
    headers = {
        'User-Agent': 'Mozilla/5.0'
    }
    
    response = requests.get(url, headers=headers)
    soup = BeautifulSoup(response.text, 'html.parser')
    
    articles = []
    for article in soup.select('article')[:10]:
        articles.append({
            'title': article.select_one('h2').text,
            'link': article.select_one('a')['href'],
            'time': article.select_one('time').text
        })
    
    return articles
```

```python
import tweepy

class TwitterFetcher:
    def __init__(self, bearer_token):
        self.client = tweepy.Client(bearer_token)

            max_results=100,
            tweet_fields=['created_at', 'text', 'public_metrics']
        )
        
        return [{
            'text': t.text,
            'created_at': t.created_at,
            'likes': t.public_metrics['like_count']
        } for t in tweets.data]
```

## 第二章：NLP情感分析

### 2.1 关键词提取
```python
import re
from collections import Counter

class KeywordExtractor:
    def __init__(self):
        self.bullish_keywords = [
            'bullish', 'buy', 'upgrade', 'surge', 'rally',
            'rise', 'gain', 'higher', 'target', 'upside',
            'safe haven', 'hedge', 'inflation', 'support'
        ]
        
        self.bearish_keywords = [
            'bearish', 'sell', 'downgrade', 'drop', 'decline',
            'fall', 'lower', 'downside', 'correction',
            'profit taking', 'risk on', 'resistance'
        ]
    
    def extract(self, text):
        text = text.lower()
        
        bullish_count = sum(1 for kw in self.bullish_keywords if kw in text)
        bearish_count = sum(1 for kw in self.bearish_keywords if kw in text)
        
        return {
            'bullish': bullish_count,
            'bearish': bearish_count,
            'score': (bullish_count - bearish_count) / max(bullish_count + bearish_count, 1)
        }
```

### 2.2 情感分析模型
```python
from transformers import pipeline

class SentimentAnalyzer:
    def __init__(self):
        # 使用金融领域模型
        self.analyzer = pipeline(
            "sentiment-analysis",
            model="ProsusAI/finbert"
        )
    
    def analyze(self, text):
        result = self.analyzer(text[:512])[0]
        
        return {
            'label': result['label'],  # positive/negative/neutral
            'score': result['score']
        }
    
    def batch_analyze(self, texts):
        return [self.analyze(t[:512]) for t in texts]
```

### 2.3 实体识别
```python
import spacy

class EntityRecognizer:
    def __init__(self):
        self.nlp = spacy.load('en_core_web_sm')
    
    def extract_entities(self, text):
        doc = self.nlp(text)
        
        entities = {
            'ORG': [],  # 组织
            'PERSON': [],  # 人
            'MONEY': [],  # 金钱
            'DATE': []  # 日期
        }
        
        for ent in doc.ents:
            if ent.label_ in entities:
                entities[ent.label_].append(ent.text)
        
        return entities
```

## 第三章：实时监控

### 3.1 定时任务
```python
import schedule
import time
from datetime import datetime

class NewsMonitor:
    def __init__(self):
        self.news = []
        self.fetcher = NewsFetcher()
        self.analyzer = SentimentAnalyzer()
    
    def job(self):
        print(f"[{datetime.now()}] 抓取新闻...")
        
        # 抓取
        news = self.fetcher.fetch_rss('reuters')
        
        # 分析
        for n in news:
            sentiment = self.analyzer.analyze(n['title'])
            n['sentiment'] = sentiment
        
        self.news.extend(news)
        print(f"  新增 {len(news)} 条")
    
    def start(self):
        # 每5分钟抓取一次
        schedule.every(5).minutes.do(self.job)
        
        while True:
            schedule.run_pending()
            time.sleep(1)
```

## 第四章：数据存储

### 4.1 MySQL存储
```python
import pymysql

class NewsDB:
    def __init__(self):
        self.conn = pymysql.connect(
            host='localhost',
            user='root',
            password='password',
            database='gold_news'
        )
    
    def save(self, news):
        with self.conn.cursor() as cursor:
            sql = """
            INSERT INTO news (title, source, sentiment, created_at)
            VALUES (%s, %s, %s, %s)
            """
            cursor.execute(sql, (
                news['title'],
                news['source'],
                news['sentiment']['label'],
                news['created_at']
            ))
        self.conn.commit()
    
    def query(self, start_time, end_time):
        with self.conn.cursor() as cursor:
            sql = """
            SELECT * FROM news 
            WHERE created_at BETWEEN %s AND %s
            ORDER BY created_at DESC
            """
            cursor.execute(sql, (start_time, end_time))
            return cursor.fetchall()
```

### 5.1 每日报告

```python
def generate_daily_report(news_list):
    bullish = [n for n in news_list if n['sentiment']['label'] == 'positive']
    bearish = [n for n in news_list if n['sentiment']['label'] == 'negative']
    
    report = f"""
    # 每日新闻报告
    
    ## 概览
    - 总新闻: {len(news_list)}
    - 利好: {len(bullish)}
    - 利空: {len(bearish)}
    
    ## 重大利好
    {chr(10).join([f"- {n['title']}" for n in bullish[:5]])}
    
    ## 重大利空
    {chr(10).join([f"- {n['title']}" for n in bearish[:5]])}
    
    ## 建议
    {'看多' if len(bullish) > len(bearish) * 2 else '看空' if len(bearish) > len(bullish) * 2 else '观望'}
    """
    
    return report
```
