## 第二章：策略实现

### 2.1 剥头皮策略
```python
class ScalpingStrategy:
    def __init__(self):
        self.target = 5  # 目标5点
        self.stop = 3    # 止损3点
        self.timeframe = '1m'
    
    def signal(self, df, delta, cum_delta):
        # 条件1: 趋势确认
        ma20 = df['close'].rolling(20).mean()
        trend_up = df['close'].iloc[-1] > ma20.iloc[-1]
        
        # 条件2: Delta确认
        delta_positive = delta > 0
        cum_delta_rising = cum_delta.iloc[-1] > cum_delta.iloc[-5]
        
        # 条件3: 突破
        high_5 = df['high'].rolling(5).max().iloc[-1]
        price_break = df['close'].iloc[-1] > high_5
        
        if trend_up and delta_positive and price_break:
            return 'LONG'
        return None
```

### 2.2 网格策略
```python
class GridStrategy:
    def __init__(self, grid_size=20, grid_count=5):
        self.grid_size = grid_size
        self.grid_count = grid_count
        self.orders = {}
    
    def place_orders(self, mid_price):
        for i in range(-self.grid_count, self.grid_count + 1):
            price = mid_price + i * self.grid_size
            
            if price not in self.orders:
                if i < 0:  # 下方挂买单
                    self.orders[price] = {
                        'type': 'BUY',
                        'price': price,
                        'status': 'pending'
                    }
                elif i > 0:  # 上方挂卖单
                    self.orders[price] = {
                        'type': 'SELL',
                        'price': price,
                        'status': 'pending'
                    }
    
    def check_filled(self, current_price):
        for price, order in self.orders.items():
            if order['status'] == 'pending':
                if order['type'] == 'BUY':
                    if current_price <= price:
                        order['status'] = 'filled'
                        self.close_position(order, current_price)
                else:
                    if current_price >= price:
                        order['status'] = 'filled'
                        self.close_position(order, current_price)
```

### 2.3 对冲策略
```python
class HedgeStrategy:
    def __init__(self):
        self.hedge_ratio = 0.5
        self.rebalance_threshold = 0.1
    
    def calculate_hedge(self, position, current_price):
        # 黄金和白银对冲
        # 假设黄金多头，白银空头
        hedge_size = position['size'] * self.hedge_ratio
        
        # 动态平衡
        gold_value = position['size'] * current_price
        silver_value = hedge_size * silver_price
        
        ratio = gold_value / silver_value
        
        if abs(ratio - 1) > self.rebalance_threshold:
            # 需要再平衡
            return True, ratio
        return False, ratio
```

```python
def order_send(symbol, order_type, volume, price=None, sl=None, tp=None):
    if price is None:
        symbol_info = mt5.symbol_info(symbol)
        if order_type == 'BUY':
            price = symbol_info.ask
        else:
            price = symbol_info.bid
    
    request = {
        "action": mt5.TRADE_ACTION_DEAL,
        "symbol": symbol,
        "volume": volume,
        "type": mt5.ORDER_TYPE_BUY if order_type == 'BUY' else mt5.ORDER_TYPE_SELL,
        "price": price,
        "sl": sl,
        "tp": tp,
        "deviation": 20,
        "magic": 234000,
        "comment": "python script",
        "type_time": mt5.ORDER_TIME_GTC,
        "type_filling": mt5.ORDER_FILLING_IOC,
    }
    
    result = mt5.order_send(request)
    return result
```

```python
def get_positions(symbol=None):
    if symbol:
        positions = mt5.positions_get(symbol=symbol)
    else:
        positions = mt5.positions_get()
    
    return [dict(p) for p in positions]

def close_position(ticket):
    position = mt5.position_get(ticket=ticket)
    
    request = {
        "action": mt5.TRADE_ACTION_DEAL,
        "symbol": position['symbol'],
        "volume": position['volume'],
        "type": mt5.ORDER_TYPE_SELL if position['type'] == 0 else mt5.ORDER_TYPE_BUY,
        "position": ticket,
        "price": mt5.symbol_info(position['symbol']).bid,
        "deviation": 20,
        "magic": 234000,
        "comment": "close",
        "type_time": mt5.ORDER_TIME_GTC,
        "type_filling": mt5.ORDER_FILLING_IOC,
    }
    
    return mt5.order_send(request)
```
