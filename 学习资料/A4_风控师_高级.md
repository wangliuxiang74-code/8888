## 第一章：风控系统架构

### 1.1 系统架构
```python
class RiskManagementSystem:
    def __init__(self):
        self.position_monitor = PositionMonitor()
        self.order_monitor = OrderMonitor()
        self.market_monitor = MarketMonitor()
        self.alert_system = AlertSystem()
    
    def check_all(self):
        """全面检查"""
        checks = {
            'position': self.position_monitor.check(),
            'order': self.order_monitor.check(),
            'market': self.market_monitor.check()
        }
        
        # 汇总
        for check_name, result in checks.items():
            if not result['pass']:
                self.alert_system.send(
                    level=result['level'],
                    message=f"{check_name}: {result['message']}"
                )
        
        return all(c['pass'] for c in checks.values())
```

## 第二章：实时风控

### 2.1 持仓风控
```python
class RealTimeRiskControl:
    def __init__(self):
        self.max_positions = 3
        self.max_single_position = 0.1  # 10%
        self.max_total_exposure = 0.2  # 20%
        self.daily_loss_limit = 0.02  # 2%
        self.single_loss_limit = 0.01  # 1%
    
    def check_position_size(self, symbol, volume, price):
        """检查仓位"""
        exposure = volume * price
        
        # 单笔限制
        if exposure > self.max_single_position:
            return False, f"单笔敞口{exposure}超过限制"
        
        # 总敞口
        current_exposure = get_total_exposure()
        if current_exposure + exposure > self.max_total_exposure:
            return False, f"总敞口超限"
        
        return True, "OK"
    
    def check_daily_loss(self, current_pnl):
        """检查日亏损"""
        if current_pnl <= -self.daily_loss_limit:
            return False, "触发日亏损熔断"
        return True, "OK"
    
    def check_position_loss(self, position):
        """检查持仓亏损"""
        pnl_pct = (position['current'] - position['entry']) / position['entry']
        
        if pnl_pct <= -self.single_loss_limit:
            return False, f"持仓亏损{pnl_pct}触发止损"
        return True, "OK"
```

class OrderRiskControl:
    def __init__(self):
        self.max_spread = 30  # 最大点差
        self.max_slippage = 5  # 最大滑点
    
    def pre_order_check(self, order):
        """下单前检查"""
        # 检查点差
        spread = get_current_spread(order['symbol'])
        if spread > self.max_spread:
            return False, f"点差{spread}超过限制"
        
        # 检查流动性
        depth = get_market_depth(order['symbol'])
        if depth < order['volume']:
            return False, "流动性不足"
        
        return True, "OK"
    
    def post_order_check(self, order, fill_price):
        """成交后检查"""
        # 检查滑点
        slippage = abs(fill_price - order['price'])
        if slippage > self.max_slippage:
            self.alert(f"滑点{slippage}超过预期")
        
        return True, "OK"

### 5.1 历史风控测试
```python
class BacktestRiskControl:
    def __init__(self, limits):
        self.limits = limits
        self.trades = []
        self.equity = [100000]  # 初始资金
    
    def on_trade(self, trade):
        """每笔交易后检查"""
        self.trades.append(trade)
        
        # 更新权益
        self.equity.append(self.equity[-1] * (1 + trade['pnl_pct']))
        
        # 检查风控
        daily_pnl = sum(t['pnl_pct'] for t in self.trades if is_today(t['time']))
        
        if daily_pnl <= -self.limits['daily_loss']:
            return 'STOP_LOSS', "日亏损超限"
        
        if len(self.trades) >= self.limits['max_trades_per_day']:
            return 'STOP_LOSS', "交易次数超限"
        
        return 'OK', ''
    
    def get_metrics(self):
        """获取风控指标"""
        returns = [e/self.equity[i]-1 for i,e in enumerate(self.equity[1:])]
        
        return {
            'max_drawdown': self.calculate_max_drawdown(),
            'sharpe_ratio': self.calculate_sharpe(returns),
            'win_rate': len([t for t in self.trades if t['pnl_pct']>0]) / len(self.trades)
        }
```
